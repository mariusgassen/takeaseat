# Take A Seat — Phase 1, Plan B: Go API Scaffold

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the foundational Go API — `openapi.yaml` contract, Chi router, Zitadel JWT validation, tenant isolation middleware, pgx DB pool, sqlc codegen setup, and a verified health endpoint.

**Architecture:** Chi 5 router with a three-layer middleware chain: JWT validation → tenant extraction (`SET LOCAL app.tenant_id`) → role enforcement. All DB access goes through pgx/v5 pool; queries generated by sqlc from hand-written SQL. The openapi.yaml at repo root is the single source of truth — orval generates TypeScript clients from it in Plan E/F.

**Tech Stack:** Go 1.26, Chi 5, pgx/v5, sqlc 1.30, Zitadel Go SDK v3, golang-migrate/v4, RFC 9457 errors

```go
// Config holds all runtime configuration loaded from environment variables.
type Config struct {

// Load reads required env vars and returns a Config.
// Returns an error if any required variable is missing.
func Load() (*Config, error) {
	c := &Config{
		PostgresURL:      os.Getenv("POSTGRES_URL"),
		RedisURL:         os.Getenv("REDIS_URL"),
		ZitadelIssuerURL: os.Getenv("ZITADEL_ISSUER_URL"),
		Port:             os.Getenv("API_PORT"),
	}

	var missing []string
	if c.PostgresURL == "" {
		missing = append(missing, "POSTGRES_URL")
	}
	if c.RedisURL == "" {
		missing = append(missing, "REDIS_URL")
	}
	if c.ZitadelIssuerURL == "" {
		missing = append(missing, "ZITADEL_ISSUER_URL")
	}
	if c.Port == "" {
		c.Port = "8000" // default
	}

	if len(missing) > 0 {
		return nil, errors.New("missing required env vars: " + join(missing))
	}
	return c, nil
}

func join(ss []string) string {
	out := ""
	for i, s := range ss {
		if i > 0 {
			out += ", "
		}
		out += s
	}
	return out
}
```

```go
package middleware

import (
	"context"
	"errors"
	"net/http"
	"strings"
	"time"

	"github.com/takeaseat/takeaseat/apps/api/internal/handler"
)

// ErrInvalidToken is returned by a TokenValidator when the token is bad.
var ErrInvalidToken = errors.New("invalid token")

// Claims holds the validated JWT claims we care about.
type Claims struct {
	Subject  string    // Zitadel user ID (JWT sub)
	TenantID string    // Zitadel org ID = our tenant UUID
	Expiry   time.Time
}

// TokenValidator validates a raw JWT string and returns claims.
// In production this hits Zitadel's JWKS endpoint.
// In tests this can be a stub.
type TokenValidator func(ctx context.Context, rawToken string) (*Claims, error)

type ctxKey int

const claimsKey ctxKey = 1

// Auth returns a middleware that validates the Bearer token and stores Claims in ctx.
// Responds 401 if the token is missing or invalid.
func Auth(validate TokenValidator) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			authHeader := r.Header.Get("Authorization")
			if authHeader == "" || !strings.HasPrefix(authHeader, "Bearer ") {
				handler.Problem(w, http.StatusUnauthorized, "missing or malformed Authorization header")
				return
			}

			rawToken := strings.TrimPrefix(authHeader, "Bearer ")
			claims, err := validate(r.Context(), rawToken)
			if err != nil {
				handler.Problem(w, http.StatusUnauthorized, "invalid token")
				return
			}

			ctx := context.WithValue(r.Context(), claimsKey, claims)
			next.ServeHTTP(w, r.WithContext(ctx))
		})
	}
}

// ClaimsFromCtx retrieves Claims from the context. Returns nil if not present.
func ClaimsFromCtx(ctx context.Context) *Claims {
	c, _ := ctx.Value(claimsKey).(*Claims)
	return c
}
```

```go
package middleware

import (
	"context"
	"fmt"
	"time"

	"github.com/zitadel/oidc/v3/pkg/oidc"
	"github.com/zitadel/oidc/v3/pkg/op"
)

const zitadelOrgIDClaim = "urn:zitadel:iam:org:id"

// NewZitadelValidator returns a TokenValidator that validates JWTs against
// Zitadel's JWKS endpoint at {issuerURL}/oauth/v2/keys.
//
// The validator:
//   1. Fetches the JWKS keyset from issuerURL (cached, refreshed on key rotation)
//   2. Verifies the JWT signature + expiry
//   3. Extracts sub (user ID) and urn:zitadel:iam:org:id (tenant ID)
func NewZitadelValidator(ctx context.Context, issuerURL string) (TokenValidator, error) {
	// Use the OIDC discovery endpoint to fetch issuer metadata (including JWKS URI).
	// The oidc package from zitadel/oidc handles JWKS caching and rotation.
	provider, err := op.NewRemoteKeySet(ctx, issuerURL+"/oauth/v2/keys")
	if err != nil {
		return nil, fmt.Errorf("init JWKS keyset for %s: %w", issuerURL, err)
	}

	return func(ctx context.Context, rawToken string) (*Claims, error) {
		// Parse + verify the JWT (signature, expiry, not-before)
		payload, err := oidc.ParseToken(rawToken, provider)
		if err != nil {
			return nil, fmt.Errorf("%w: %v", ErrInvalidToken, err)
		}

		// Extract our tenant ID from the Zitadel org claim
		orgID, ok := payload.Claims[zitadelOrgIDClaim].(string)
		if !ok || orgID == "" {
			return nil, fmt.Errorf("%w: missing %s claim", ErrInvalidToken, zitadelOrgIDClaim)
		}

		return &Claims{
			Subject:  payload.Subject,
			TenantID: orgID,
			Expiry:   time.Time(payload.Expiration),
		}, nil
	}, nil
}
```

```go
import (
	"context"
	"fmt"
	"net/http"

	"github.com/jackc/pgx/v5/pgxpool"
	"github.com/takeaseat/takeaseat/apps/api/internal/handler"

// Tenant runs after Auth. It:
//  1. Reads TenantID from the JWT claims in context
//  2. Acquires a pgx connection from the pool
//  3. Executes SET LOCAL app.tenant_id = $1 (scopes RLS to this tenant)
//  4. Stores the connection in context for handlers to use
//
// The connection is released (returned to pool) when the request finishes.
func Tenant(pool *pgxpool.Pool) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			claims := ClaimsFromCtx(r.Context())
			if claims == nil || claims.TenantID == "" {
				handler.Problem(w, http.StatusUnauthorized, "tenant not resolved — auth middleware must run first")
				return
			}

			if pool == nil {
				handler.Problem(w, http.StatusInternalServerError, "database pool not configured")
				return
			}

			conn, err := pool.Acquire(r.Context())
			if err != nil {
				handler.Problem(w, http.StatusInternalServerError, "failed to acquire DB connection")
				return
			}
			defer conn.Release()

			// Scope all queries in this request to the tenant's rows via RLS.
			if _, err := conn.Exec(r.Context(),
				fmt.Sprintf("SET LOCAL app.tenant_id = '%s'", claims.TenantID),
			); err != nil {
				handler.Problem(w, http.StatusInternalServerError, "failed to set tenant context")
				return
			}

			ctx := contextWithConn(r.Context(), conn)
			next.ServeHTTP(w, r.WithContext(ctx))
		})
	}
}

type connKey int

const dbConnKey connKey = 1

func contextWithConn(ctx context.Context, conn *pgxpool.Conn) context.Context {
	return context.WithValue(ctx, dbConnKey, conn)
}

// ConnFromCtx retrieves the scoped DB connection from context.
// Returns nil if not present (e.g., in tests or unauthenticated routes).
func ConnFromCtx(ctx context.Context) *pgxpool.Conn {
	c, _ := ctx.Value(dbConnKey).(*pgxpool.Conn)
	return c
}
```

```go
package middleware

import (
	"context"
	"net/http"

	"github.com/takeaseat/takeaseat/apps/api/internal/handler"
)

// Role mirrors the user_role DB enum.
type Role string

const (
	RoleGuest   Role = "guest"
	RoleMember  Role = "member"
	RoleManager Role = "manager"
	RoleAdmin   Role = "admin"
)

var roleLevels = map[Role]int{
	RoleGuest:   0,
	RoleMember:  1,
	RoleManager: 2,
	RoleAdmin:   3,
}

// Level returns a numeric rank (higher = more privileged).
func (r Role) Level() int { return roleLevels[r] }

// Satisfies returns true if this role meets or exceeds the required role.
func (r Role) Satisfies(required Role) bool { return r.Level() >= required.Level() }

type roleKey int

const userRoleKey roleKey = 1

// ContextWithRole stores the current user's role in context (set by RequireRole middleware).
func ContextWithRole(ctx context.Context, r Role) context.Context {
	return context.WithValue(ctx, userRoleKey, r)
}

// RoleFromCtx retrieves the current user's role. Returns RoleGuest if not set.
func RoleFromCtx(ctx context.Context) Role {
	r, _ := ctx.Value(userRoleKey).(Role)
	if r == "" {
		return RoleGuest
	}
	return r
}

// RequireRole returns middleware that enforces a minimum role.
// It reads the user's role from context (put there by a previous LookupRole middleware)
// and responds 403 if the role is insufficient.
//
// Usage: mount LookupRole first, then RequireRole on specific routes.
func RequireRole(minimum Role) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			current := RoleFromCtx(r.Context())
			if !current.Satisfies(minimum) {
				handler.Problem(w, http.StatusForbidden,
					"insufficient role: requires "+string(minimum)+", have "+string(current))
				return
			}
			next.ServeHTTP(w, r)
		})
	}
}
```

Note: `LookupRole` (which fetches the user's actual role from DB and stores it in context) is implemented in Plan C alongside the users handler, since it requires DB queries. For Plan B, `RequireRole` reads from context — it's a no-op if role isn't stored yet.

```go
package handler

import (
	"encoding/json"
	"net/http"

	"github.com/jackc/pgx/v5/pgxpool"
)

// Health returns an http.Handler for GET /health.
// It pings the DB pool if non-nil; returns {"status":"ok"} on success.
func Health(pool *pgxpool.Pool) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		if pool != nil {
			if err := pool.Ping(r.Context()); err != nil {
				Problem(w, http.StatusServiceUnavailable, "database unavailable: "+err.Error())
				return
			}
		}
		w.Header().Set("Content-Type", "application/json")
		w.WriteHeader(http.StatusOK)
		_ = json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
	})
}
```
